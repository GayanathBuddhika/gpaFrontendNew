import { CompanyService } from './../../../service/company.service';
import { Company } from './../../../model/company';
import { PlanService } from './../../../service/plan.service';
import { SubscriptionsDetails } from './../../../model/subscriptions';
import { SubscriptionService } from './../../../service/subscription.service';
import { FormGroup, FormBuilder, Validators } from '@angular/forms';
import { Component, OnInit, Input, OnDestroy } from '@angular/core';
import { Plan } from 'src/app/model/plan';
import { BsDatepickerConfig } from 'ngx-bootstrap/datepicker/public_api';
import { subscribeOn } from 'rxjs/operators';
import { DatePipe } from '@angular/common';
import { ToastrManager } from 'ng6-toastr-notifications';
import { ConstantPool } from '@angular/compiler';
import { Observable, Subscription } from 'rxjs';

@Component({
  selector: 'app-subscription-plan',
  templateUrl: './subscription-plan.component.html',
  styleUrls: ['./subscription-plan.component.css']
})
export class SubscriptionPlanComponent implements OnInit, OnDestroy {

  subscriptionPlanForm: FormGroup;
  subscriptionPlanList: Subscription[] = [];

  private plan: Plan;
  // --- Subscription 
  private subscription: Subscription = new Subscription();

  selectedComapnyForPlan: Company;
  // ---  for Get the Subscription plan when selecte the plan
  selectedSubscription: SubscriptionsDetails;
  // --- for disable the ngx DatePicker
  dateDisable: boolean = true;

  // --- for get the current dates
  today = new Date();
  // --- for get the end date for selected plan
  endDateForSelectPlan: Date;
  // --- for view the plan datials after select the start date( when create the starte date this will be a true)
  viewPlanDeatials: boolean = false;

  // --- for get the EndDateForSelectPlan as yyyy-mm-dd format
  formatEndDateForSelectPlan: string;
  // --- for get the startDate as yyyy-mm-dd format
  formatStartDate: string;
  // --- for get the saved plan 
  savedplan: Plan;



  datePigureConfig: Partial<BsDatepickerConfig>;

  @Input() selectedplan: Plan;
  @Input() edit: boolean = false;
  @Input() endDate: string;
  @Input() enable: boolean;
  constructor(
    private frm: FormBuilder,
    private subscriptionService: SubscriptionService,
    private planService: PlanService,
    private companyService: CompanyService,
    // to convert the Date to yyyy-mm-dd format
    private datePipe: DatePipe,
    public toster: ToastrManager


  ) { }

  ngOnInit() {
    console.log("selected paln----", this.selectedplan);
    console.log("today ----", this.today);
    console.log("in plan add", this.endDate);

    this.datePigureConfig = Object.assign({},
      {
        dateInputFormat: 'YYYY-MM-DD',
        minDate: new Date(this.endDate),
      })

    this.subscriptionPlanForm = this.frm.group({
      company: [""],
      subscription: ["", Validators.required],
      startDate: [new Date().toJSON()],
      endDate: [""],
      enable: [""],
      date: [""]
    });

    this.getAllSubscription();

    // --- add subscribe methode to the subscription object
    this.subscription.add(

      this.companyService._selectCompanyForPlan.subscribe(data => {
        console.log("on subscription-plan ", data);
        this.selectedComapnyForPlan = data;
        console.log("on subscription-plan  .................", this.selectedComapnyForPlan);

      })
    );

  }

  // --- convenience getter for easy access to form fields
  get f() {
    return this.subscriptionPlanForm.controls;
  }

  ngAfterViewInit() {
    if (this.edit) {
      this.subscriptionPlanForm.get('subscription').patchValue(this.selectedplan.subscription);
      this.subscriptionPlanForm.get('company').patchValue(this.selectedplan.company);
      this.subscriptionPlanForm.get('startDate').patchValue(this.selectedplan.startDate);
    }
  }


  ngOnDestroy() {
    // --- unsubscribe all subscribe methode 
    this.subscription.unsubscribe();
  }


  getAllSubscription() {


    // --- add subscribe methode to the subscription object
    this.subscription.add(

      this.subscriptionService.listSubscription().subscribe(data => {
        this.subscriptionPlanList = data;
        console.log(this.subscriptionPlanList);
      }, err => {
        console.log(err);
      })
    );

  }

  onsubmit() {

    this.plan = this.subscriptionPlanForm.value;

    if (!this.edit) {
      // this.subscriptionPlanForm.get('company').patchValue(this.selectedComapnyForPlan);
      // console.log("add new subscription",this.subscriptionPlanForm.value);

      this.plan.company = this.selectedComapnyForPlan;
      console.log("add new subscription", this.subscriptionPlanForm.get('startDate').value);
      console.log("add new subscription", this.plan);

    } else {

      this.plan.id = this.selectedplan.id;
      this.plan.ai = this.selectedplan.ai;
      this.plan.enable = this.selectedplan.enable;

    }

    console.log("plan befor saved", this.plan);

    // --- add subscribe methode to the subscription object
    this.subscription.add(

      this.planService.savePlan(this.plan, this.edit).subscribe(data => {
        this.savedplan = data;
        console.log("saved plan", this.savedplan);
        if (this.edit !== true) {
          this.toster.successToastr("Success", "Successfully saved", { position: "bottom-right", animate: "slideFromBottom" });
          this.planService._set_ngxModal_add(true);
          this.planService._addPlanToList.next(this.savedplan);
        } else {
          this.toster.successToastr("Success", "Successfully edited", { position: "bottom-right", animate: "slideFromBottom" });
          this.planService._set_ngxmodal_edit(true);
          this.planService._editPlanToList.next(this.savedplan);
        }
        console.log(data);
      }, err => {
        console.log(err);
        if (this.edit !== true) {
          this.planService._set_ngxModal_add(true);
        } else {
          this.planService._set_ngxmodal_edit(false);
        }

        this.toster.errorToastr(err, "Error", { position: "bottom-right", animate: "slideFromBottom" })

      })
    );




  }

  compareplan(s1: SubscriptionsDetails, s2: SubscriptionsDetails) {
    return s1.id === s2.id;
  }

  onSelectedSubscription(subscription: SubscriptionsDetails) {

    this.selectedSubscription = subscription;
    this.dateDisable = false;

  }

  onSelectedStartDate() {

    const plan2: Plan = this.subscriptionPlanForm.value;
    plan2.company = this.selectedComapnyForPlan;
    console.log(plan2.startDate);
    const validMonth: number = this.selectedSubscription.validMonths;

    // --- add subscribe methode to the subscription object
    this.subscription.add(
      // --- for get the end date by back-end.
      this.planService.getEndDate(validMonth, plan2).subscribe(data => {
        console.log(data);
        const plan3: Plan = data;
        this.formatEndDateForSelectPlan = plan3.endDate;
        this.formatStartDate = plan3.startDate;

      }, err => {
        console.log(err);
      })
    );


    this.viewPlanDeatials = true;




  }



}
