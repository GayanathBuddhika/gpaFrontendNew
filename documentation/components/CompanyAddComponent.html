import { BussinessType } from './../../../model/bussinessType';

import { CompanyService } from '../../../service/company.service';

import { Observable, Subscription } from 'rxjs';


@Component({
  selector: 'app-company-add',
  templateUrl: './company-add.component.html',
  styleUrls: ['./company-add.component.css']
})
export class CompanyAddComponent implements OnInit, OnDestroy {
  private businessTypes: BussinessType;

  // --- Subscription 
  private subscription: Subscription = new Subscription();

  private company: Company;

  // --- this create for assing company object after create the new company
  savedCompany: Company;

  private plan: Plan;

  subscriptionList: SubscriptionsDetails[];

  // ---  for add the defolt value for the department input field
  dname: string = "Administration";

  // ---  for add the defolt value for the branch input field
  bname: string;

  // --- for default branch
  private branch: Branch = new Branch();

  // --- for get the saved default branch
  savedbranch: Branch;

  // --- for get the saved default department
  savedDepartment: Department;

  // --- for default department
  private department: Department = new Department();

  // --- for create the deptobranch list
  depToBranch: DepToBranch[] = [];

  private depToBranchObjet: DepToBranch = new DepToBranch();

  // ---  get value from parent componentcompanyLlist.component.html   
  @Input() edit = false;

  // ---  get value from parent componentcompanyLlist.component.html   
  @Input() selectcompany: Company;

  constructor(
    private companyService: CompanyService,
    private bussinessTypeService: BussinessTypeService,
    private subscriptionService: SubscriptionService,
    private planService: PlanService,
    private modalservice: BsModalService,
    public toster: ToastrManager,
    private departmentService: DepartmentService,
    private branchService: BranchService,
    private depToBranchService: DepartmentAssingToBranchService
  ) { }

  ngOnInit() {
    this.getBusinessTypes();

  }

  ngOnDestroy() {
    // --- unsubscribe all subscribe methode 
    this.subscription.unsubscribe();
  }



  // --- get bussiness type
  getBusinessTypes() {
    this.bussinessTypeService.getBusinessType().subscribe(data => {
      console.log(data);
      this.businessTypes = data;

    }, err => {
      console.log(err);
    })
  }


  // --- get subscriptionList
  getSubscription() {
    // --- add subscribe methode to the subscription object
    this.subscription.add(

      this.subscriptionService.listSubscription().subscribe(data => {
        this.subscriptionList = data;
      })

    );
   
  }



  // --- save and edit company methode
  saveCompany(form: NgForm) {

    this.company = form.value;
    console.log(this.bname);
    console.log(this.dname);

    // --- if user edit Company, this.edit==true and assing selected company id and ai to company id and ai
    if (this.edit) {
      this.company.id = this.selectcompany.id;
      this.company.ai = this.selectcompany.ai;
      this.company.edit = this.edit;

    }

     // --- add subscribe methode to the subscription object
     this.subscription.add(
       
    // pass the branch name , department name, and edit value with the company object to create the default comany and branch 
    this.companyService.addCompany(this.company, this.bname, this.dname).subscribe(data => {
      // --- assing saved company to savedComp
      // --- this saved company object return from the back end after save the copany
      this.savedCompany = data.company;
      console.log('saved company', this.savedCompany);

      // --- for identyfy create new company or edit object by front end
      if (this.edit !== true) {
        // --- for identyfy create new company by back end
        if (data.action === 'saved') {
          this.toster.successToastr("Success", "Successfully saved", { position: "bottom-right", animate: "slideFromBottom" });
          this.companyService._set_ngxModal_add(true);
          this.companyService._addCompanyToList.next(this.savedCompany);

        } else {

          this.toster.errorToastr("Error", "company already exists add", { position: "bottom-right", animate: "slideFromBottom" });

        }

      } else {
        // --- for identyfy editd company by back end
        if (data.action === 'edited') {
          this.toster.successToastr("Success", "Successfully edited", { position: "bottom-right", animate: "slideFromBottom" });
          this.companyService._set_ngxModal_edit(true);
          this.companyService._editCompanyToList.next(this.savedCompany);
        } else {
          console.log("edit company", data.action);
          this.toster.errorToastr("Error", "company already exists edit", { position: "bottom-right", animate: "slideFromBottom" });
        }

      }
    }, err => {

      if (this.edit !== true) {
        this.companyService._set_ngxModal_add(true);
      } else {
        this.companyService._set_ngxModal_edit(false);
      }

      this.toster.errorToastr(err, "Error", { position: "bottom-right", animate: "slideFromBottom" });
    })

     );



  }


}
