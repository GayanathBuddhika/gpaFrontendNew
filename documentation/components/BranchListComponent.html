import { DepartmentAssingToBranchService } from './../../../service/department-assing-to-branch.service';
import { DepartmentService } from 'src/app/service/department.service';
import { Component, OnInit, TemplateRef } from '@angular/core';
import { BsModalService, BsModalRef } from 'ngx-bootstrap/modal';
import { Branch } from 'src/app/model/branch';
import { Subject, Subscription } from 'rxjs';
import { BranchService } from 'src/app/service/branch.service';
import { CompanyService } from 'src/app/service/company.service';
import { Company } from 'src/app/model/company';
import {TableModule} from 'primeng/table';
import { ToastrManager } from 'ng6-toastr-notifications';
import { ConfirmationService } from 'primeng/api';
import { THIS_EXPR } from '@angular/compiler/src/output/output_ast';

@Component({
  selector: 'app-branch-list',
  templateUrl: './branch-list.component.html',
  styleUrls: ['./branch-list.component.css']
})
export class BranchListComponent implements OnInit {

   // --- Subscription 
   private subscription: Subscription = new Subscription();
   
  // --- modalRef_edit to refer the modal (ngx-bootstrap) which is in branch-list.component.ts
  modalRef_edit: BsModalRef;

  // --- modalRef_assign to refer the modal (ngx-bootstrap) which is in branch-list.component.ts
  modalRef_assign: BsModalRef;

  id: String;
  branchList: Branch[];  
  branch: Branch;
  companyId:string;
  branchId: Branch;
  headers:any[];
  clickItem =[];
  bId: Branch;
  
  constructor( 
    private branchService: BranchService,
    private modalService: BsModalService,
    private companyService: CompanyService,
    private departmentService: DepartmentService,
    private departmentAssingToBranchService: DepartmentAssingToBranchService,
    private confirmationService: ConfirmationService,
    public toastr: ToastrManager) { }

  ngOnInit() {
    this.getCompanyId(); 
    
  
    // ---This is for headers in the BRANCH table
    this.headers =
    [
      {field: 'no', header: 'No'},
      { field: 'name', header: 'Name'},
      { field: 'address', header: 'Address'},
      { field: 'edit', header: 'Edit'},
      { field: 'delete', header: 'Delete'},
      { field: 'assign', header: 'Assign'},
    ];


     // ---if add or triggers, make them add into the lIst using subject.
     this.subscription.add(
      this.branchService.addBranchToList_$().subscribe(res=>{            
        let bList = [...this.branchList];         
        bList.unshift(res)
        this.branchList = bList; 
       
      }))
     

    // ---if edit or triggers, make them add into the lIst using subject.
    this.subscription.add(
      this.branchService.editBranchToList_$().subscribe(res=>{
        let itemIndex = this.branchList.findIndex(item => item.id == res.id);
        this.branchList[itemIndex] = res;
        } ))

    this.subscription.add(  
      this.branchService.editBranchToList_$().subscribe(res=>{
       let itemIndex = this.branchList.findIndex(item => item.id == res.id);
       this.branchList[itemIndex] = res;
      } ))

    // --- Here subscribe the _ngxmodel_branch_edit to get next value 
    // --- if it is true then close the model
    this.subscription.add(
     this.branchService.ngxmodel_branch_edit_$().subscribe(data=>{
      if (data === true) {
        this.modalRef_edit.hide();
    }
    }))

    this.subscription.add(
    this.departmentAssingToBranchService.ngxmodel_assign_$().subscribe(data=>{
      if(data === true){
        this.modalRef_assign.hide();
      }
    }))
  }

  ngOnDestroy() {
    this.subscription.unsubscribe();
  }

  
  openModal1(template1: TemplateRef<any>) {
    this.modalRef_edit = this.modalService.show(template1,{class: 'modal-lg'});
  }

  openModal2(template2: TemplateRef<any>) {
    this.modalRef_assign = this.modalService.show(template2);
  }

  // --- get company id from company list dropdown
  getCompanyId(){
    this.subscription.add(
    this.companyService.getComId_$().subscribe(res => {
      this.companyId =res.id;
      this.getBranchList();
   }))
  }

  // --- get information for delete a branch
  deleteBranch(branchId){ 
      
    // ---This will ask to confirm befor delete the value
       this.confirmationService.confirm({
        message: 'Are you sure that you want to proceed?',
        header: 'Confirmation',
        icon: 'pi pi-exclamation-triangle',
        accept: () => {
          this.subscription.add(
            this.branchService.deleteBranch(branchId.id).subscribe(res=>{
            // ---when delete the branch real time update branch-list
            let index=this.branchList.indexOf(branchId)
            this.branchList = this.branchList.filter((val,i) => i!=index); 
            this.toastr.successToastr("Success","Data is deleted",{position:"bottom-right",animate:"slideFromBottom"}) 
           },err=>{
            this.toastr.errorToastr(JSON.parse(JSON.stringify(err)).error.message,"Error",{position:"bottom-right",animate:"slideFromBottom"})
           }))
        },
        reject: () => {

        }
    });

    // this.branchService.deleteBranch(branchId.id).subscribe(
    //   data => {
    // }, err => {
    //   console.log(err);
    // })
  }  

  // --- get  branch list
  getBranchList(){
    this.subscription.add(
    this.branchService.getBranch(this.companyId).subscribe(branchData=>{
      this.branchList= branchData;
    }, err => {
      this.toastr.errorToastr(JSON.parse(JSON.stringify(err)).error.message,"Error",{position:"bottom-right",animate:"slideFromBottom"})
    }))
  }

  // --- update branch list
  update(branch: Branch, template_edit){
    this.branch=branch;
    this.openModal1(template_edit);
  }

  // --- assign department to branch
  assignDept(branchId: Branch,template_assign){
    this.branch=branchId;
    this.departmentService._set_branch(this.branch);
    this.openModal2(template_assign);
  }

  // ---get ckecked value from front end for delete
  onChange(event:any,branch:Branch){
    if (event.target.checked) {

      this.clickItem.push(branch);
    } 
    if (!event.target.checked) {

      let index = this.clickItem.indexOf(branch);

      if (index > -1) {

        this.clickItem.splice(index, 1);
      }
    }
    
  }

  // ---This is delete all selected branches
  deleteAll(){
    this.confirmationService.confirm({
      message: 'Are you sure that you want to proceed?',
      header: 'Confirmation',
      icon: 'pi pi-exclamation-triangle',
      accept: () => {
        this.subscription.add(
          this.branchService.deleteAll(this.clickItem).subscribe(res=>{
            for(var i=0;i<this.clickItem.length;i++){
              let index=this.branchList.indexOf(this.clickItem[i]);
              this.branchList = this.branchList.filter((val,i) => i!=index); 
            }
          this.toastr.successToastr("Success","Data is deleted",{position:"bottom-right",animate:"slideFromBottom"}) 
         },err=>{
          this.toastr.errorToastr(JSON.parse(JSON.stringify(err)).error.message,"Error",{position:"bottom-right",animate:"slideFromBottom"})
         }))
      },
      reject: () => {

      }
  });
    
  }
}
