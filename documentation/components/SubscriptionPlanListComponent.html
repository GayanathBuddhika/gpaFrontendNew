import { CompanyService } from './../../../service/company.service';
import { Company } from 'src/app/model/company';
import { TemplateRef } from '@angular/core';
import { BsModalRef, BsModalService } from 'ngx-bootstrap/modal';
import { PlanService } from './../../../service/plan.service';
import { Component, OnInit, Input, OnDestroy } from '@angular/core';
import { Plan } from 'src/app/model/plan';
import { Observable, Subscription } from 'rxjs';

@Component({
  selector: 'app-subscription-plan-list',
  templateUrl: './subscription-plan-list.component.html',
  styleUrls: ['./subscription-plan-list.component.css']
})
export class SubscriptionPlanListComponent implements OnInit, OnDestroy {

  headers: any[];
  planList: Plan[];
  startDate: string;
  endDate: string;
  enable: boolean;

  // --- Subscription 
  private subscription: Subscription = new Subscription();

  modalRef: BsModalRef;

  selectPlan: Plan;

  subsEdit: boolean = false;

  addbutton: boolean = true;



  @Input() selectedComId: string;


  constructor(private palnService: PlanService, private modalService: BsModalService) { }

  ngOnInit() {

    this.headers =
      [
        { field: 'no', header: 'No' },
        { field: 'plan_Name', header: 'Plan Name' },
        { field: 'start_date', header: 'Start Date' },
        { field: 'end_date', header: 'End Date' },
        { field: 'isEnabled', header: 'status' },
        { field: 'edit', header: 'Edit' }
      ];

    this.getAllPlan();
    // --- add subscribe methode to the subscription object
    this.subscription.add(
      this.palnService.get_ngxModal_edit_$().subscribe(data => {
        if (data === true) {
          this.modalRef.hide();
        }
      })
    );


    // --- add subscribe methode to the subscription object
    this.subscription.add(
      this.palnService._addPlanToList.subscribe(data => {
        let pList = [...this.planList];
        pList.unshift(data);
        this.planList = pList;
      })
    );


    // --- add subscribe methode to the subscription object
    this.subscription.add(
      this.palnService._editPlanToList.subscribe(data => {
        let index = this.planList.findIndex(plan => plan.id === data.id);
        this.planList[index] = data;
      })
    );



  }

  ngOnDestroy() {
    // --- unsubscribe all subscribe methode 
    this.subscription.unsubscribe();
  }

  getAllPlan() {

    // --- add subscribe methode to the subscription object
    this.subscription.add(
      this.palnService.findAllplanByComId(this.selectedComId).subscribe(data => {
        this.planList = data;


        if (this.planList.length == 1 || this.planList.length == 0) {
          this.addbutton = false;
        }

        if (this.planList.length == 1) {

          this.startDate = this.planList[0].startDate;
          this.endDate = this.planList[0].endDate;
          this.enable = this.planList[0].enable;

          console.log("start date", this.startDate);
          console.log("end data", this.endDate);
          console.log("end data", this.planList[0].enable);



        }


        console.log("plan data", data);

      }, err => {
        console.log(err);
      })
    );




  }

  edit(plan: Plan, tempalte) {
    this.selectPlan = plan;
    this.openModal(tempalte);
    this.subsEdit = true;
  }

  openModal(template: TemplateRef<any>) {
    this.modalRef = this.modalService.show(template,{class: 'modal-lg'});
  }

}
