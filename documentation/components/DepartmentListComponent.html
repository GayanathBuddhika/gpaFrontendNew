import { BranchService } from 'src/app/service/branch.service';
import { Component, OnInit, TemplateRef, ViewEncapsulation } from '@angular/core';
import { Department } from 'src/app/model/department';
import { Subject, Subscription } from 'rxjs';
import { DepartmentService } from 'src/app/service/department.service';
import { BsModalService, BsModalRef } from 'ngx-bootstrap/modal';
import { CompanyService } from 'src/app/service/company.service';
import {TableModule} from 'primeng/table';
import { ConfirmationService } from 'primeng/api';
import { ToastrManager } from 'ng6-toastr-notifications';


@Component({
  selector: 'app-department-list',
  templateUrl: './department-list.component.html',
  styleUrls: ['./department-list.component.css'],
  encapsulation: ViewEncapsulation.None
})
export class DepartmentListComponent implements OnInit {

  // --- Subscription 
  private subscription: Subscription = new Subscription();

  // --- modalRef_add to refer the modal (ngx-bootstrap) which is in user.component.ts
  modalRef_edit: BsModalRef;
  departmentList: Department [];
  dtOptions: DataTables.Settings = {};
  dtTrigger: Subject<any>= new Subject();
  deleteMsg: Object;
  companyId: string;
  department:Department;
  headers:any[];
  departmentId:Department;
  clickItem =[];

  constructor(
    private departmentService: DepartmentService,
    private modalService: BsModalService,
    private companyService: CompanyService,
    private branchService:BranchService,
    private confirmationService: ConfirmationService,
    public toastr: ToastrManager) { }

  ngOnInit() {
    this.getCompanyId();
    // this.branchService.ngx_model_val_branch_$().subscribe(data=>{
    //   if (data === true) {
    //     this.modalRef_edit.hide();
    // }
    // });
    this.headers =
    [ {field: 'no', header: 'No'},
      { field: 'name', header: 'Name'},
      { field: 'edit', header: 'Edit'},
      { field: 'delete', header: 'Delete'}
    ];

     // ---if add  triggers, make them add into the lIst using subject.
     this.subscription.add(
     this.departmentService.addDeptToList_$().subscribe(res=>{            
      let dList = [...this.departmentList];         
      dList.unshift(res)
      this.departmentList = dList; 
    }))

    // ---if edit  triggers, make them add into the lIst using subject.
    this.subscription.add(
    this.departmentService.editDeptToList_$().subscribe(res=>{
      let itemIndex = this.departmentList.findIndex(item => item.id == res.id);
      this.departmentList[itemIndex] = res;
      } ))

    // --- Here subscribe the _ngxmodel_dept_edit to get next value 
    // --- if it is true then close the model
    this.subscription.add(
    this.departmentService.ngxmodel_dept_edit_$().subscribe(data=>{
      if (data === true) {
        this.modalRef_edit.hide();
    }
    }))
  }

  ngOnDestroy() {
    this.subscription.unsubscribe();
}
  // --- get companyId from compani list drodown
  getCompanyId(){
    this.subscription.add(
    this.companyService.getComId_$().subscribe(res => {
    this.companyId =res.id;
    this.getDepartment();
   }))
  }

  // --- get list of department by company id
  getDepartment(){
    this.subscription.add(
    this.departmentService.getDepartment(this.companyId).
    subscribe(departmentData=>{
    this.departmentList = departmentData;
    this.dtTrigger.next();
    }, err => {
    console.log(err);
    }))
  }

  openModal(template: TemplateRef<any>) {
    this.modalRef_edit = this.modalService.show(template,{class: 'modal-lg'});
  }

  // ---get the department id for delate department values
  deleteDepartment(departmentId){
    this.delete(departmentId);
  }

  // --- Delete depatrtment
  delete(department){

    // ---This will ask to confirm befor delete the value
       this.confirmationService.confirm({
        message: 'Are you sure that you want to proceed?',
        header: 'Confirmation',
        icon: 'pi pi-exclamation-triangle',
        accept: () => {
          this.subscription.add(
            this.departmentService.deleteDepartment(department.id).subscribe(res=>{
            // ---when delete the department real time update department-list
            let index=this.departmentList.indexOf(department)
            this.departmentList = this.departmentList.filter((val,i) => i!=index); 
            this.toastr.successToastr("Success","Data is deleted",{position:"bottom-right",animate:"slideFromBottom"}) 
           },err=>{
            this.toastr.errorToastr(JSON.parse(JSON.stringify(err)).error.message,"Error",{position:"bottom-right",animate:"slideFromBottom"})
           }))
        },
        reject: () => {

        }
    });
    // this.departmentService.deleteDepartment(department.id).subscribe(
    // data => {console.log(data)
    // }, err => {
    // console.log(err);
    // })
  }

  // --- update Department
  update(department: Department, tem){
    this.department=department;
    this.openModal(tem);
  }
  
  // ---get ckecked value from front end for delete
  onChange(event:any,dept:Department){
    if (event.target.checked) {

      this.clickItem.push(dept);
    } 
    if (!event.target.checked) {

      let index = this.clickItem.indexOf(dept);

      if (index > -1) {

        this.clickItem.splice(index, 1);
      }
    }
    
  }

  // ---This is delete all selected branches
  deleteAll(){
    this.confirmationService.confirm({
      message: 'Are you sure that you want to proceed?',
      header: 'Confirmation',
      icon: 'pi pi-exclamation-triangle',
      accept: () => {
        this.subscription.add(
          this.departmentService.deleteAll(this.clickItem).subscribe(res=>{
            for(var i=0;i<this.clickItem.length;i++){
              let index=this.departmentList.indexOf(this.clickItem[i]);
              this.departmentList = this.departmentList.filter((val,i) => i!=index); 
            } 
          this.toastr.successToastr("Success","Data is deleted",{position:"bottom-right",animate:"slideFromBottom"}) 
         },err=>{
          this.toastr.errorToastr(JSON.parse(JSON.stringify(err)).error.message,"Error",{position:"bottom-right",animate:"slideFromBottom"})
         }))
      },
      reject: () => {

      }
  });
    
  }
}
